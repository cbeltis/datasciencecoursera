days <- weekdays(activity_data$date)
day_type <- vector()
for (i in 1:nrow(activity_data)) {
  if (days[i] == "Saturday") {day_type[i] <- "Weekend"} 
  else if (days[i] == "Sunday") {day_type[i] <- "Weekend"} 
  else {day_type[i] <- "Weekday"}}
